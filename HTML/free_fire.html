using UnityEngine; public class PlayerController : MonoBehaviour { public float speed = 5f; public float rotationSpeed = 720f; void Update() { float horizontal = Input.GetAxis("Horizontal"); float vertical = Input.GetAxis("Vertical"); Vector3 direction = new Vector3(horizontal, 0f, vertical).normalized; if (direction.magnitude >= 0.1f) { float targetAngle = Mathf.Atan2(direction.x, direction.z) * Mathf.Rad2Deg; float angle = Mathf.SmoothDampAngle(transform.eulerAngles.y, targetAngle, ref rotationSpeed, 0.1f); transform.rotation = Quaternion.Euler(0f, angle, 0f); transform.Translate(Vector3.forward * speed * Time.deltaTime); } } }
